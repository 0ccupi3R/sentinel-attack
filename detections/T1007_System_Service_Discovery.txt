// Name: System Service Discovery
// Description: Checks for execution of MITRE ATT&CK T1007
//
// Severity: High
//
// QueryFrequency: 1h
//
// QueryPeriod: 1h
//
// AlertTriggerThreshold: 1
//
// DataSource: #Sysmon
//
// Tactics: #Discovery
//
SysmonOSSEM
| where EventID == 1 and (process_name contains "net.exe" 
or process_name contains "tasklist.exe" 
or process_name contains "sc.exe" 
or process_name contains "wmic.exe") 
and (process_command_line contains "*net* start*" 
or process_command_line contains "*tasklist \/svc*" 
or process_command_line contains "*sc* query*" 
or process_command_line contains "wmic* service where*")
