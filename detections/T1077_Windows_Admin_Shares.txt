// Name: Windows Admin Shares - Network
// Description: Checks for execution of MITRE ATT&CK T1077
//
// Severity: High
//
// QueryFrequency: 1h
//
// QueryPeriod: 1h
//
// AlertTriggerThreshold: 1
//
// DataSource: #Sysmon
//
// Tactics: #Lateral Movement
//
SysmonParser
| where EventID == 3 and Image contains "net.exe"
and (CommandLine contains "use"
or CommandLine contains "session"
or CommandLine contains "file")
