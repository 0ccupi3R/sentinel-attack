// Name: Remote System Discovery - Network
// Description: Checks for execution of MITRE ATT&CK T1018
//
// Severity: High
//
// QueryFrequency: 1h
//
// QueryPeriod: 1h
//
// AlertTriggerThreshold: 1
//
// DataSource: #Sysmon
//
// Tactics: #Discovery
//
SysmonOSSEM 
| where event_id == 1 and(process_name contains "net.exe"
and process_command_line contains "config") 
or (process_name contains "ipconfig.exe"
or process_name contains "netsh.exe" 
or process_name contains "arp.exe"
or process_name contains "nbtstat.exe")
